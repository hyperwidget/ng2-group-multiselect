Ng2-Multi-Select

An Angular 2 component that allows the selection of multiple items from a select style dropdown box. It allows the entered values to be organized by groups to allow for easier sorting, and selection of all items in a group.

Check it out on github

With Grouping

Selected Values in multiselect:
{{demoVal.name}}{{isLast ? '' : ', '}}
  • Markdown
  • Controller
                                
        <hyper-multiselect 
            [inbound]="demo2" 
            (outbound)="refreshDemo2Values($event)" 
            [displayKey]="'name'"
            [groupBy]="'type'"
            [allSelected]="true"> 
        </hyper-multiselect>
                            
                            
                                
        this.demo2 = [{{'{'}} id: 1, name: "Rose Tyler", type: "companion" }, 
            {{'{'}} id: 2, name: "Donna Noble", type: "companion" }, 
            {{'{'}} id: 3, name: "Martha Jones", type: "companion" }, 
            {{'{'}} id: 4, name: "Dalek", type: "enemy" }
        ];

            
        public refreshDemo2Values(value: any): void {{'{'}} 
            this.demo2Vals = value; 
        }
